
#02 Removes duplicate characters from a string, preserving first occurrence order.


def remove_duplicates(s: str) -> str:
    if not isinstance(s, str):
        raise TypeError("Input must be a string.")
    seen = set()
    out_chars = []
    for ch in s:
        if ch not in seen:
            seen.add(ch)
            out_chars.append(ch)
    return ''.join(out_chars)
